declare @ID int, @OrderNum int, @Linked bit, @TimeIn smalldatetime, @ServerID smallint,
        @NumOfGuests tinyint, @Opened bit, @TableName varchar(20), @PickupTime varchar(8),
        @PhoneNum varchar(10), @PickupPerson varchar(20), @TakeOut bit, @DailyOrdNum smallint,
        @BillPrinted bit, @Attention bit, @NewID int

declare curs_table cursor for 
  select ID, OrderNum, Linked, TimeIn, ServerID, NumOfGuests, Opened, TableName, PickupTime,
         PhoneNum, PickupPerson, TakeOut, DailyOrdNum, BillPrinted, Attention
  from tblTable
  where TakeOut = 1
    and ID <= 200
  order by ID

  open curs_table
  fetch next from curs_table into @ID, @OrderNum, @Linked, @TimeIn, @ServerID,
                                  @NumOfGuests, @Opened, @TableName, @PickupTime,
                                  @PhoneNum, @PickupPerson, @TakeOut, @DailyOrdNum,
                                  @BillPrinted, @Attention
  set @NewID = 201
  while (@@fetch_status = 0) begin

    update tblTable
    set OrderNum = @OrderNum, 
        Linked = @Linked, 
        TimeIn = @TimeIn, 
        ServerID = @ServerID,
        NumOfGuests = @NumOfGuests, 
        Opened = @Opened, 
        TableName = @TableName, 
        PickupTime = @PickupTime,
        PhoneNum = @PhoneNum, 
        PickupPerson = @PickupPerson, 
        TakeOut = @TakeOut, 
        DailyOrdNum = @DailyOrdNum,
        BillPrinted = @BillPrinted, 
        Attention = @Attention
    where ID = @NewID

    update tblTableLayout
    set TableID = @NewID
    where TableID = @ID

    set @NewID = @NewID + 1

    fetch next from curs_table into @ID, @OrderNum, @Linked, @TimeIn, @ServerID,
                                    @NumOfGuests, @Opened, @TableName, @PickupTime,
                                    @PhoneNum, @PickupPerson, @TakeOut, @DailyOrdNum,
                                    @BillPrinted, @Attention
  end
  close curs_table
deallocate curs_table'
GO

EXEC dbo.sp_executesql @statement = N'
if not exists (select * 
               from tblPayment) begin

  insert into tblPayment (SalesID, PaymentTypeID, PaidAmount, TipAmount)
  select s.ID, 1, s.CashPaid, s.CashTipPaid
  from tblSales s
  where s.OriginalTransType in (0, 1)
    and s.CashPaid <> 0
  order by s.ID

  insert into tblPayment (SalesID, PaymentTypeID, PaidAmount, TipAmount)
  select s.ID, 1, 0, s.CashTipPaid
  from tblSales s
  where s.OriginalTransType in (0, 1)
    and s.CashPaid = 0
    and s.CashTipPaid <> 0
  order by s.ID

  insert into tblPayment (SalesID, PaymentTypeID, PaidAmount, TipAmount)
  select s.ID, 2, s.DebitPaid, s.DebitTipPaid
  from tblSales s
  where s.OriginalTransType in (0, 1)
    and s.DebitPaid <> 0
  order by s.ID

  insert into tblPayment (SalesID, PaymentTypeID, PaidAmount, TipAmount)
  select s.ID, 2, 0, s.DebitTipPaid
  from tblSales s
  where s.OriginalTransType in (0, 1)
    and s.DebitPaid = 0
    and s.DebitTipPaid <> 0
  order by s.ID

  insert into tblPayment (SalesID, PaymentTypeID, PaidAmount, TipAmount)
  select s.ID, 3, s.VisaPaid, s.CreditTipPaid
  from tblSales s
  where s.OriginalTransType in (0, 1)
    and s.VisaPaid <> 0
  order by s.ID

  insert into tblPayment (SalesID, PaymentTypeID, PaidAmount, TipAmount)
  select s.ID, 4, s.McPaid, case when s.VisaPaid <> 0 then 0 else s.CreditTipPaid end
  from tblSales s
  where s.OriginalTransType in (0, 1)
    and s.McPaid <> 0
  order by s.ID

  insert into tblPayment (SalesID, PaymentTypeID, PaidAmount, TipAmount)
  select s.ID, 5, s.AmexPaid, case when (s.VisaPaid <> 0) or (s.McPaid <> 0) then 0 else s.CreditTipPaid end
  from tblSales s
  where s.OriginalTransType in (0, 1)
    and s.AmexPaid <> 0
  order by s.ID

  insert into tblPayment (SalesID, PaymentTypeID, PaidAmount, TipAmount)
  select s.ID, 3, 0, s.CreditTipPaid
  from tblSales s
  where s.OriginalTransType in (0, 1)
    and s.VisaPaid = 0
    and s.DebitPaid = 0
    and s.AmexPaid = 0
    and s.CreditTipPaid <> 0
  order by s.ID

  insert into tblPayment (SalesID, PaymentTypeID, PaidAmount, TipAmount)
  select s.ID, 6, s.CouponPaid, 0
  from tblSales s
  where s.OriginalTransType in (0, 1)
    and s.CouponPaid <> 0
  order by s.ID

end'
GO

if exists (select *
           from sys.indexes
           where name = ''idx_CustomerNum'' 
             and object_id = object_id(''dbo.tblCustomer'')
             and is_unique_constraint = 0) begin

  if not exists (select CustomerNum
                 from tblCustomer
                 group by CustomerNum
                 having count(CustomerNum) > 1) begin

    CREATE UNIQUE NONCLUSTERED INDEX [idx_CustomerNum] ON [dbo].[tblCustomer]
    (
	    [CustomerNum] ASC
    )WITH (PAD_INDEX = OFF, STATISTICS_NORECOMPUTE = OFF, SORT_IN_TEMPDB = OFF, IGNORE_DUP_KEY = OFF, DROP_EXISTING = ON, ONLINE = OFF, ALLOW_ROW_LOCKS = ON, ALLOW_PAGE_LOCKS = ON) ON [PRIMARY]

  end
  else begin

    declare @CustomerNum varchar(15), @OriginalID int, @EarnedPoints int

    declare duplicate_customer_num cursor for
      select CustomerNum, min(ID) as OriginalID, sum(EarnedPoints) as EarnedPoints
      from tblCustomer
      group by CustomerNum
      having count(CustomerNum) > 1

      open duplicate_customer_num
      fetch next from duplicate_customer_num into @CustomerNum, @OriginalID, @EarnedPoints
      while (@@fetch_status = 0) begin

        update s
        set s.CustomerID = @OriginalID
        --select *
        from tblSales s
          inner join tblCustomer c on s.CustomerID = c.ID
        where c.CustomerNum = @CustomerNum
          and c.ID <> @OriginalID

        update p
        set p.CustomerID = @OriginalID
        --select *
        from tblPrepaidCards p
          inner join tblCustomer c on p.CustomerID = c.ID
        where c.CustomerNum = @CustomerNum
          and c.ID <> @OriginalID

        update b
        set b.CustomerID = @OriginalID
        --select *
        from tblBillCollectHistory b
          inner join tblCustomer c on b.CustomerID = c.ID
        where c.CustomerNum = @CustomerNum
          and c.ID <> @OriginalID

        update c
        set c.CustomerNum = replicate(''0'', 15 - len(convert(varchar, c.ID))) + convert(varchar, c.ID)
        from tblCustomer c
        where c.CustomerNum = @CustomerNum
          and c.ID <> @OriginalID

        update c
        set c.EarnedPoints = @EarnedPoints
        from tblCustomer c
        where c.ID = @OriginalID

        fetch next from duplicate_customer_num into @CustomerNum, @OriginalID, @EarnedPoints
      end
      close duplicate_customer_num
    deallocate duplicate_customer_num

    if not exists (select CustomerNum
                   from tblCustomer
                   group by CustomerNum
                   having count(CustomerNum) > 1) begin

      CREATE UNIQUE NONCLUSTERED INDEX [idx_CustomerNum] ON [dbo].[tblCustomer]
      (
	      [CustomerNum] ASC
      )WITH (PAD_INDEX = OFF, STATISTICS_NORECOMPUTE = OFF, SORT_IN_TEMPDB = OFF, IGNORE_DUP_KEY = OFF, DROP_EXISTING = ON, ONLINE = OFF, ALLOW_ROW_LOCKS = ON, ALLOW_PAGE_LOCKS = ON) ON [PRIMARY]

    end

  end

end'

           from tblSalesDetail 
           where ItemName = '') begin

  update sd
  set sd.ItemName = case when (i.Taste = 1 and sd.SizeID > 1) then z.SizeName else i.IName end,
      sd.ItemName2 = isnull(i.IName2, ''),
      sd.SizeName = case when i.Taste = 1 then '' else z.SizeName end
  from tblSalesDetail sd
    inner join tblItem i on sd.ItemID = i.ID
    inner join tblSize z on sd.SizeID = z.ID 
  where sd.ItemName = ''

end
