EXEC dbo.sp_executesql @statement = N'

           from sys.objects 
           where object_id = OBJECT_ID(N''[dbo].[tmpUpdate_ApplyDSCItem]'') 
             and type in (N''U'')) begin

  set a.ApplyGeneralDSC = 0
  from tblAvailableSize a

  from tblAvailableSize a, tmpUpdate_ApplyDSCItem i

'

EXEC dbo.sp_executesql @statement = N'

           from sys.objects 
           where object_id = OBJECT_ID(N''[dbo].[tmpUpdate_DayRateDiscount]'') 
             and type in (N''U'')) begin

  declare @ItemID smallint, @SizeID tinyint, @MonRate float, @TueRate float, @WedRate float, 
          @ThuRate float, @FriRate float, @SatRate float, @SunRate float

  declare curs_day_rate cursor for
    select a.ItemID, a.SizeID, u.MondayRate, u.TuesdayRate, u.WednesdayRate, 
           u.ThursdayRate, u.FridayRate, u.SaturdayRate, u.SundayRate
    from tblAvailableSize a, tmpUpdate_DayRateDiscount u
    where a.ItemID = u.ItemID

    open curs_day_rate
    fetch next from curs_day_rate into @ItemID, @SizeID, @MonRate, @TueRate, @WedRate, @ThuRate, @FriRate, @SatRate, @SunRate

      insert into tblDayHourDiscount (ItemID, SizeID, WeekDayIndex, ApplyDay, ApplyDayRate, ApplyDayPrice, ApplyHour,
